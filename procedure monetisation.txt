        class SubscriptionManager {
            constructor() {
                this.checkSubscriptionStatus();
            }

            checkSubscriptionStatus() {
                // Vérifier le statut d'abonnement (localStorage pour la démo)
                const subscription = localStorage.getItem('premiumSubscription');
                if (subscription) {
                    const subData = JSON.parse(subscription);
                    if (new Date(subData.expiresAt) > new Date()) {
                        this.activatePremium();
                        return;
                    }
                }
                
                // Vérifier les paramètres URL pour le retour de Stripe
                const urlParams = new URLSearchParams(window.location.search);
                if (urlParams.get('success') === 'true') {
                    this.handleSuccessfulPayment();
                }

                this.unlockAllGames();

                // Simuler l'activation premium (en production, vérifiez via webhook)
                const expirationDate = new Date();
                expirationDate.setMonth(expirationDate.getMonth() + 1);
                
                localStorage.setItem('premiumSubscription', JSON.stringify({
                    plan: 'premium',
                    expiresAt: expirationDate.toISOString(),
                    activatedAt: new Date().toISOString()
                }));
                
                this.activatePremium();
